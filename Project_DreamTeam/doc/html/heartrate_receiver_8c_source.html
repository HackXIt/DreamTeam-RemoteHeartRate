244 void HrReceiver(void *argument) {
245  uint32_t flags_return;
246  HRRECEIVER_RETVAL receiver_return;
247  while(1) {
248  flags_return = osThreadFlagsWait(I2C_EVENT, osFlagsWaitAny, osWaitForever);
249  if(flagErrorHandler(flags_return)) {
250  // On flag error, continue loop from beginning
251  continue;
252  }
253  if(flags_return & I2C_EVENT) {
254  receiver_return = hrReceiver_handle_heartrate();
255  if(hrReceiver_failureHandler(receiver_return)) {
256  printf_("HRRECEIVER: Error handling I2C event: 0x%02x%s", receiver_return, newLine);
257  }
258  }
259  }
