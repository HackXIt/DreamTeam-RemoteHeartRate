46 void Parser(void *argument) {
47  parser_module_t module = *(parser_module_t*)argument;
48  uint32_t flags = 0;
49  PARSER_RETVAL ret;
50  while(1) {
51  flags = osThreadFlagsWait(NEW_INPUT|MSG_EVENT|COPY_REATTEMPT, osFlagsWaitAny, osWaitForever);
52  if(flagErrorHandler(flags)) {
53  // On flag error, continue loop from beginning
54  continue;
55  }
56  if(flags & NEW_INPUT) {
57  ret = parser_handle_input();
58  if(parser_failureHandler(ret)) {
59  printf_("PARSER: error parsing input: 0x%02x%s", ret, newLine);
60  }
61  }
62  if(flags & MSG_EVENT) {
63  ret = parser_handle_message();
64  if(parser_failureHandler(ret)) {
65  printf_("PARSER: error parsing message: 0x%02x%s", ret, newLine);
66  }
67  }
68  if(flags & COPY_REATTEMPT) {
69  ret = parser_handle_copyReattempt();
70  if(parser_failureHandler(ret)) {
71  printf_("PARSER: error handling copy re-attempt: 0x%02x%s", ret, newLine);
72  }
73  }
74  }
75 }

104 PARSER_RETVAL parser_handle_message() {

106  // Request semaphore to lock input
107  if((parser_return = acquire_semaphore(msgSem, MODULE_SEM_TIMEOUT)) != PARSER_OK) {
108  return parser_return;

117  // -------------------------------- END OF PARSE

119  // TODO Notify webserver about network data (if any)

124 #ifdef USE_WEBSERVER
