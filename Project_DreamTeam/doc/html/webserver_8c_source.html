46 void Webserver(void *argument) {
47  webserver_module_t module = *(webserver_module_t*)argument;
48  uint32_t flags = 0;
49  WEBSERVER_RETVAL ret;
50  while(1) {
51  flags = osThreadFlagsWait(NEW_CONNECTION|NEW_REQUEST, osFlagsWaitAny, osWaitForever);
52  if(flagErrorHandler(flags)) {
53  // On flag error, continue loop from beginning
54  continue;
55  }
56  if(flags & NEW_CONNECTION) {
57  ret = webserver_handle_connection();
58  if(webserver_failureHandler(ret)) {
59  printf_("WEBSERVER: error parsing input: 0x%02x%s", ret, newLine);
60  }
61  }
62  if(flags & NEW_REQUEST) {
63  ret = webserver_handle_request();
64  if(webserver_failureHandler(ret)) {
65  printf_("WEBSERVER: error parsing input: 0x%02x%s", ret, newLine);
66  }
67  }
68  }
69 }
