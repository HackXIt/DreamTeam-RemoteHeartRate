59 void ConsoleMonitor(void *argument) {
60  console_module_t *module = (console_module_t*)argument;
61  // switch-case state-machine handling thread flags
62  uint32_t flags = 0;
63  CONSOLE_RETVAL ret;
64  while(1) {
65  // Wait for thread flags
66  flags = osThreadFlagsWait(RX_BYTE|BUFFER_OVERRUN, osFlagsWaitAny, osWaitForever);
67  if(flagErrorHandler(flags)) {
68  // On flag error, continue loop from beginning
69  continue;
70  }
71  // Console (USART2) State-Machine
72  if(flags & RX_BYTE) {
73  ret = console_handle_rxByte();
74  if(console_failureHandler(ret)) {
75  printf_("CONSOLE: error handling received byte: 0x%02x%s", ret, newLine);
76  } else if (ret == CONSOLE_NEW_DATA) {
77  ret = console_handle_newInput();
78  if(console_failureHandler(ret)) {
79  printf_("CONSOLE: error handling new input: 0x%02x%s", ret, newLine);
80  }
81  uart2BufferIndex = 0;
82  }
83  }
84  if(flags & BUFFER_OVERRUN) {
85  ret = console_handle_bufferOverrun();
86  if(console_failureHandler(ret)) {
87  printf_("CONSOLE: error handling buffer overrun: 0x%02x%s", ret, newLine);
88  }
89  }
90  }
91 }

95 CONSOLE_RETVAL console_handle_rxByte() {
96  HAL_StatusTypeDef ret;
97  uint32_t flag;
98  char c = (char)rx_byte;
99 
100  ret = HAL_UART_Transmit(&huart2, (uint8_t*)&c, 1, HAL_MAX_DELAY); // UART line echo
101  if(halStatusHandler(ret)) {
102  printf_("CONSOLE: echo transmit error: ");
103  translate_UART_Error(&huart2);
104  }
105 
106  if(c == '\r') {
107  ret = HAL_UART_Transmit(&huart2, (uint8_t*)newLine, 2, HAL_MAX_DELAY);
108  if(halStatusHandler(ret)) {
109  printf_("CONSOLE: newline transmit error: ");
110  translate_UART_Error(&huart2);
111  }
112  return CONSOLE_NEW_DATA;
113  } else {
114  uart2Buffer[uart2BufferIndex++] = c;
115  if(uart2BufferIndex >= CONSOLE_UART_BUFFER_SIZE) {
116  flag = osThreadFlagsSet(consoleThreadId, BUFFER_OVERRUN);
117  if(flagErrorHandler(flag)) {
118  printf_("CONSOLE: error setting flag: 0x%02x%s", ret, newLine);
119  }
120  }
121  }
122  return CONSOLE_OK;
123 }

136 CONSOLE_RETVAL console_handle_newInput() {
137  uint32_t flag_return;
138  CONSOLE_RETVAL console_return;
139  if((console_return = console_enter_copy()) != CONSOLE_OK) {
140  return console_return;
141  }
142  // -------------------------------- START OF COPY
143  // Request heap memory for data
144  input = pvPortMalloc(sizeof(char)*uart2BufferIndex+1);
145  if(input == NULL) {
146  return CONSOLE_MEM_ERROR;
147  }
148  // Copy data
149  memcpy(input, uart2Buffer, uart2BufferIndex);
150  input[uart2BufferIndex] = '\0';
151  // -------------------------------- END OF COPY
152  if((console_return = release_semaphore(inputSem)) != CONSOLE_OK) {
153  return console_return;
154  }
155  // Notify parser about new input
156  flag_return = osThreadFlagsSet(parserThreadId, NEW_INPUT);
157  if(flagErrorHandler(flag_return)) {
158  return CONSOLE_FAILED_FLAG;
159  }
160  return CONSOLE_OK;
161 }
