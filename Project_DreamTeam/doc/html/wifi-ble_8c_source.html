106 void WifiReceiver(void *argument) {
107  uint32_t flags;
108  static size_t old_pos = WIFIBLE_UART_BUFFER_SIZE; // Track the position of last character processed
109  size_t reattempt_pos1 = 0;
110  size_t reattempt_pos2 = 0;
111  WIFIBLE_RETVAL ret;
112  while(1) {
113  // Wait for thread flags
114  flags = osThreadFlagsWait(IDLE_EVENT|COPY_REATTEMPT|USER_INPUT|NEW_RESPONSE, osFlagsWaitAny, osWaitForever);
115  if(flagErrorHandler(flags)) {
116  // On flag error, continue loop from beginning
117  continue;
118  }
119  // Wifi Receiver (USART1) State-Machine
120  if(flags & IDLE_EVENT) {
121  /*
122  HAL_DMA_StateTypeDef dma_state = HAL_DMA_GetState(&hdma_usart1_rx);
123  if(dma_state != HAL_DMA_STATE_BUSY) {
124  printf_("WIFIBLE: DMA stopped: gState == 0x%08lx RxState == 0x%08lx%s", huart1.gState, huart1.RxState, newLine);
125  }
126  */
127  size_t new_pos = WIFIBLE_UART_BUFFER_SIZE - huart1.hdmarx->Instance->CNDTR; // Compute the new position in the buffer
128  if(new_pos != old_pos) // Check if any new data is received
129  {
130  ret = wifible_handle_newData(old_pos, new_pos);
131  if(wifible_failureHandler(ret) && ret == WIFIBLE_FAILED_COPY) {
132  osThreadFlagsSet(parserThreadId, COPY_REATTEMPT);
133  reattempt_pos1 = old_pos;
134  reattempt_pos2 = new_pos;
135  }
136  old_pos = new_pos; // Update the position of the last character processed
137  }
138  }
139  if(flags & COPY_REATTEMPT) {
140  ret = wifible_handle_newData(reattempt_pos1, reattempt_pos2);
141  if(wifible_failureHandler(ret)) {
142  printf_("WIFIBLE: 2nd copy-attempt failed: 0x%02x%s", ret, newLine);
143  }
144  reattempt_pos1 = 0;
145  reattempt_pos2 = 0;
146  }
147  if(flags & USER_INPUT) {
148  ret = wifible_handle_userInput();
149  if(wifible_failureHandler(ret)) {
150  printf_("WIFIBLE: Handling user input failed: 0x%02x%s", ret, newLine);
151  }
152  }
153  if(flags & NEW_RESPONSE) {
154  ret = wifible_handle_newRequest();
155  if(wifible_failureHandler(ret)) {
156  printf_("WIFIBLE: Handling new request failed: 0x%02x%s", ret, newLine);
157  }
158  }
159  }

304 WIFIBLE_RETVAL wifible_handle_newData(size_t old_pos, size_t new_pos) {
305  size_t length;
306 #ifdef WIFIBLE_DEBUG
307  HAL_StatusTypeDef ret;
308 #endif
309  // The transmissions here are in blocking mode, otherwise output is cutoff!
310  // If received data is BIGGER than buffer size, then output will be cutoff! (=> Increase buffer size)
311  // The function does not abort on transmission error, but will abort on copy-error
312  if (new_pos > old_pos) // If data does not wrap around the buffer
313  {
314  length = new_pos - old_pos;
315  // Transmit data
316 #ifdef WIFIBLE_DEBUG
317  ret = HAL_UART_Transmit(&huart2, &uart1Buffer[old_pos], length, HAL_MAX_DELAY);
318  if(halStatusHandler(ret)) {
319  printf_("WIFIBLE: ");
320  translate_UART_Error(&huart2);
321  }
322 #endif
323  // Process your data => uart1Buffer[old_pos] TO uart1Buffer[old_pos+length] == Received DATA
324  return wifible_attempt_fullCopy(old_pos, length);
325  }
326  else // If data wraps around the buffer
327  {
328  // If you process data in here, you'll need to partially construct your data
329 
330  // First transmit the data until the end of the buffer
331  length = WIFIBLE_UART_BUFFER_SIZE - old_pos;
332  size_t tmp = length;
333 #ifdef WIFIBLE_DEBUG
334  ret = HAL_UART_Transmit(&huart2, &uart1Buffer[old_pos], length, HAL_MAX_DELAY);
335  if(halStatusHandler(ret)) {
336  printf_("WIFIBLE: ");
337  translate_UART_Error(&huart2);
338  }
339 #endif
340  // Then transmit the remaining data from the beginning of the buffer
341  length = new_pos;
342 #ifdef WIFIBLE_DEBUG
343  ret = HAL_UART_Transmit(&huart2, uart1Buffer, length, HAL_MAX_DELAY);
344  if(halStatusHandler(ret)) {
345  printf_("WIFIBLE: ");
346  translate_UART_Error(&huart2);
347  }
348 #endif
349  // Process data partially
350  return wifible_attempt_partialCopy(old_pos, tmp, length);
351  }

364  if(acquire_semaphore(msgSem, MODULE_SEM_TIMEOUT) != WIFIBLE_OK) {

366  }

373 WIFIBLE_RETVAL wifible_attempt_fullCopy(size_t start_pos, size_t length) {
374  uint32_t flag_return;
375  WIFIBLE_RETVAL wifible_return;
376  if((wifible_return = wifible_enter_copy()) != WIFIBLE_OK) {
377  return wifible_return;
378  }
379  // -------------------------------- START OF COPY
380  // Request new heap memory for received data
381  newMsg = pvPortMalloc(sizeof(char)*length+1);
382  if(newMsg == NULL) {
383  return WIFIBLE_MEM_ERROR;
384  }
385  // Copy data
386  memcpy(newMsg, uart1Buffer+start_pos, length);
387  newMsg[length] = '\0';
388  // -------------------------------- END OF COPY
389  if((wifible_return = release_semaphore(msgSem)) != WIFIBLE_OK) {
390  return wifible_return;
391  }
392  // Notify parser about new message
393  flag_return = osThreadFlagsSet(parserThreadId, MSG_EVENT);
394  if(flagErrorHandler(flag_return)) {
395  return WIFIBLE_FAILED_FLAG;
396  }
397  return WIFIBLE_OK;
398 }
