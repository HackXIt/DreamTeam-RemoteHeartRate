20 /***
21  * translate_I2C_Error provides a detailed analysis of the I2C Error
22  * and prints it to the terminal via the printf_ function (see printf_.c & printf_.h)
23  * @param hi2c the I2C-Interface to be analyised
24  */
25 void translate_I2C_Error(I2C_HandleTypeDef *hi2c) {
26  uint32_t error_value = HAL_I2C_GetError(hi2c);
27  printf_("I2C: ");
28  // NOTE: Some errors need to be cleared by software
29  switch(error_value) {
30  case HAL_I2C_ERROR_NONE:
31  printf_("No error");
32  break;
33  case HAL_I2C_ERROR_BERR:
34  printf_("BERR error");
35  LL_I2C_ClearFlag_BERR(hi2c->Instance);
36  break;
37  case HAL_I2C_ERROR_ARLO:
38  printf_("ARLO error");
39  LL_I2C_ClearFlag_ARLO(hi2c->Instance);
40  break;
41  case HAL_I2C_ERROR_AF:
42  printf_("ACKF error");
43  break;
44  case HAL_I2C_ERROR_OVR:
45  printf_("OVR error");
46  break;
47  case HAL_I2C_ERROR_DMA:
48  printf_("DMA transfer error");
49  break;
50  case HAL_I2C_ERROR_TIMEOUT:
51  printf_("Timeout error");
52  break;
53  case HAL_I2C_ERROR_SIZE:
54  printf_("Size Management error");
55  break;
56  case HAL_I2C_ERROR_DMA_PARAM:
57  printf_("DMA Parameter error");
58  break;
59  default:
60  printf_("Multiple I2C errors");
61  if(error_value & HAL_I2C_ERROR_BERR) LL_I2C_ClearFlag_BERR(hi2c->Instance);
62  if(error_value & HAL_I2C_ERROR_ARLO) LL_I2C_ClearFlag_ARLO(hi2c->Instance);
63  break;
64  }
65  printf_(" (0x%08lx)\r\n", error_value);
66 }

119 /***
120  * MAX30101_read_sample reads a single sample of values (currently only red-led values)
121  * from the sensor's FIFO
122  * @param sensor The hardware address of the sensor's FIFO the sample shall be retrieved from
123  * @param sample A pointer to the MAX30101_SAMPLE struct the data can be stored in
124  * @return Returns GET_REG_SUCCESS if a sample was succeessfully retrieved from the sensor's FIFO
125  * return GET_REG_FAILED if reading the FIFO resulted in a failure;
126  */
127 INIT_STATUS MAX30101_read_sample(MAX30101 *sensor, MAX30101_SAMPLE *sample)
128 {
129  uint8_t reg_data[(ACTIVE_LED_CHANNELS * 3)] = { 0 };
130 
131  if(HAL_I2C_Mem_Read(sensor->i2c_handle, MAX30101_I2C_READ, FIFO_DATA, I2C_MEMADD_SIZE_8BIT, (uint8_t *)reg_data, sizeof(reg_data), I2C_TIMEOUT) != HAL_OK) return GET_REG_FAILED;
132  else
133  {
134  uint32_t tmp_value = 0;
135  if ( ACTIVE_LED_CHANNELS >= 1)
136  {
137  tmp_value |= (reg_data[0] << 16);
138  tmp_value |= (reg_data[1] << 8);
139  tmp_value |= (reg_data[2]);
140  sample->red_led_value = tmp_value;
141  sample->tick_time = HAL_GetTick();
142  }
143 
144  sample->tick_time = HAL_GetTick();
145  }
146  return GET_REG_SUCCESS;
147 }
